And if you look at ~~//Tools/eqsubst.ML, the first thing that happen is:

(* changes object "=" to meta "==" which prepares a given rewrite rule *)
fun prep_meta_eq ctxt =
  Simplifier.mksimps ctxt #> map Drule.zero_var_indexes;
