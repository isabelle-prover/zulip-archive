   apply (induction x)

or:

proof (induction x)
  (*click on the skeleton produced by Isabelle/jEdit*)
   case 0
  then show ?case
    sledgehammer
   sorry
next
  case (Suc n)
  then show ?case
    sledgehammer
   sorry
qed
