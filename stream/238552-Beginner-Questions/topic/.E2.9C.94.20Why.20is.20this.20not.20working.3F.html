Kevin Kappelmann (Jul 15 2026 at 14:55):

like this:

lemma conj_swapI: "A ⟹ B ⟹ B ∧ A" by auto
lemmas conj_intros = conjI conj_swapI

notepad
begin
  assume "A" "B"
  from ‹A› ‹B› have "A ∧ B" by (rule conj_intros) (*OK*)
  from ‹B› ‹A› have "A ∧ B" by (rule conj_intros) (*OK*)
end
