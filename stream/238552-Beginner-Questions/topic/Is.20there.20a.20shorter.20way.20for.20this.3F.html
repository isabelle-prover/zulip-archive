lemma blubb:
  assumes "∀i ≤ card S + 1. ∀i' ≤ card S + 1.
          i ≠ i' ⟶ (f::nat⇒'a) i ≠ f i'"
  shows "n ≤ card S + 1 ⟹ n ≤ card (⋃i≤n. {f i})"
proof (induction n)
