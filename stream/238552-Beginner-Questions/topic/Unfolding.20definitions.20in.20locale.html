Mathias Fleury (Nov 08 2023 at 05:58):

locale test =
  fixes a :: nat
  assumes X: "a = 3"

definition (in test) mycst where
  ‹mycst = a * a›

lemma (in test) a_squared: "a *a = 9"
  using X by auto


lemma
  fixes a :: nat
  assumes "a = 3"
  shows "a*a=9"
proof -
  interpret test a
    apply unfold_locales
    apply (use assms in auto)
    done
  show ?thesis
    by (rule a_squared)
qed
