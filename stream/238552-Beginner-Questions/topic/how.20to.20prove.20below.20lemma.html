lemma length_ins_sort: "length (ins_sort xs) = length xs"
proof (induction xs)
  case Nil
  then show ?case by auto
next
  case (Cons a ys)
  then show ?case apply auto
  proof(induct ys)
    case Nil
    then show ?case apply auto done
  next
    case (Cons b zs)
    then show ?case
    proof (cases "a ≤ b")
      case True
      then show ?thesis  apply simp  sorry
    next
      case False
      then show ?thesis sorry
    qed
  qed
qed
