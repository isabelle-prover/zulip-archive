val tl = toplevel()

//print(st.goal)
assert(st.goal.length==1)
val pre = st.goal.head.asInstanceOf[QRHLSubgoal].pre
assert(pre.toString == "‚Ñ≠ùî©ùîû[x1 = y] ‚äì ‚ä§")
pre.checkWelltyped(tl.state.isabelle, GIsabelle.predicateT)
