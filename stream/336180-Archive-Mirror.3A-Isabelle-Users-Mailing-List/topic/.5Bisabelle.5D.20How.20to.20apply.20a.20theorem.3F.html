theory test2 imports Cardinal

definition
"Emedding(embed, small, big) \<equiv> small \<union> (big - embed``(small))"

lemma emmbed_prop: assumes "embed \<in> inj(small, big)"
shows
"big \<approx> Emedding(embed, small, big)" and
