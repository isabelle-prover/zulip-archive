theory Collatz
imports Parity

definition
collatzNext :: "nat \<Rightarrow> nat"
where
"collatzNext n = (if even n then (n div 2) else 3*n + 1)"

lemma "0<n \<Longrightarrow> 0 < collatzNext n"
apply (auto simp add: collatzNext_def)
